## Technical Design

### 1. Create Subscription Hook

```typescript
// src/frontend/src/hooks/useSubscription.ts
import { useAuthStore } from '../store';
import { subscriptionApi } from '../services/subscriptionApi';

export function useSubscription() {
  const { user } = useAuthStore();
  
  const canAccessGame = async (gameId: string): Promise<boolean> => {
    if (!user) return false;
    
    try {
      const response = await subscriptionApi.checkGameAccess(user.id, gameId);
      return response.hasAccess;
    } catch (error) {
      console.error('Subscription check failed:', error);
      return false;
    }
  };
  
  return { canAccessGame };
}
```

### 3. Add to Each Game

```typescript
// Pattern to add to each game component
import { useSubscription } from '../hooks/useSubscription';
import { AccessDenied } from '../components/ui/AccessDenied';

export const MyGame = memo(function MyGameComponent() {
  const { canAccessGame } = useSubscription();
  const [hasAccess, setHasAccess] = useState<boolean | null>(null);
  
  useEffect(() => {
    const checkAccess = async () => {
      const access = await canAccessGame('game-id');
      setHasAccess(access);
    };
    checkAccess();
  }, [canAccessGame]);
  
  if (hasAccess === false) {
    return <AccessDenied game="Game Name" />;
  }
  
  if (hasAccess === null) {
    return <LoadingState />;
  }
  
  // ... rest of game
});
```
